public static String signThousands(String str){ String resultStr = ""; String decimalStr =""; if(isNum(str)){ if(str.contains(".")){ decimalStr = str.substring(str.indexOf(".")); str=str.substring(0,str.indexOf(".")); } List tempStrList = new ArrayList(); while(str.length()>3){ String tempStr=","+str.substring(str.length()-3); tempStrList.add(tempStr); str=str.substring(0,str.length()-3); } resultStr+=str; for(int i=tempStrList.size();i>0;i--){ resultStr+=tempStrList.get(i-1); } resultStr+=decimalStr; str = resultStr; } return str;
